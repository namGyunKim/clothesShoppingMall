create table cmember(
    id varchar2(50),
    password varchar2(50),
    tel varchar2(15),
    address varchar2(100),
    grade varchar2(10),
    answer varchar2(50),
    constraint cmember__pk primary key (id)
);

create table clothes(
    id number generated by default as identity,
    title varchar2(50),
    content varchar2(50),
    price number(10),
    kate varchar2(50),
    constraint clothes__pk primary key(id)
);

create table clothesG(
    id number(10),
    title varchar2(50),
    content varchar2(50),
    price number(10),
    kate varchar2(50),
    constraint clothesG__pk primary key(id)
);

create table clothesB(
    id number generated by default as identity,
    clothesid number(10),
    userid varchar2(50),
    title varchar2(50),
    content varchar2(50),
    price number(10),
    kate varchar2(50),
    count number(10),
    constraint clothesB__pk primary key(id)
);

create table payrecord(
    id number generated by default as identity,
    constraint payrecord__pk primary key (id),
    userid varchar2(50),
    title varchar2(200),
    price number(10),
    paydate date default sysdate
);
